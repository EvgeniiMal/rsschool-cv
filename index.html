function businessNamesAndRatings(apiResult ) {
  var len = apiResult.results.length;
  var arr=[];
  for (var i = 0; i < len; i++) {
    var rating = apiResult.results[i].rating;
    var name = apiResult.results[i].name;
    arr.push({rating, name });
  }
  return arr;
}
